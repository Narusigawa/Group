using System;
using System.Drawing;
using System.Windows.Forms;

namespace cond
{
    public partial class MainForm : Form
    {
        private HeaderControl headerControl;
        private MenuControl menuControl;
        private Panel pnlContent;
        private CatalogControl catalogControl;

        private User currentUser;          // авторизованный пользователь

        public MainForm()
        {
            InitializeComponent();
            SetupMainForm();
        }

        private void SetupMainForm()
        {
            this.Text = "Кондитерская Sweet Delights";
            this.WindowState = FormWindowState.Maximized;
            this.MinimumSize = new Size(900, 650);
            this.BackColor = ThemeColors.Background;

            headerControl = new HeaderControl();
            headerControl.MenuClick += ToggleMenu;
            headerControl.ProfileClick += OpenProfile;
            headerControl.CartClick += OpenCart;
            this.Controls.Add(headerControl);

            menuControl = new MenuControl();
            menuControl.NavigationRequested += Navigate;
            this.Controls.Add(menuControl);
            menuControl.BringToFront();

            pnlContent = new Panel
            {
                Dock = DockStyle.Fill,
                AutoScroll = true,
                BackColor = ThemeColors.Background
            };
            this.Controls.Add(pnlContent);

            catalogControl = new CatalogControl();
            catalogControl.AddToCartClicked += OnAddToCart;
            catalogControl.ProductClicked += OnProductClicked;   // подписка на клик по карточке
            catalogControl.LoadProducts();
            ShowPage(catalogControl);
        }

        // Обработка добавления товара в корзину
        private void OnAddToCart(object sender, Product product)
        {
            if (currentUser == null)
            {
                MessageBox.Show("Чтобы добавить товар в корзину, войдите в систему.",
                                "Требуется авторизация",
                                MessageBoxButtons.OK,
                                MessageBoxIcon.Information);
                return;
            }

            bool result = DatabaseHelper.AddToCart(currentUser.Id, product.Id, 1);
            if (result)
                MessageBox.Show($"Товар \"{product.Name}\" добавлен в корзину!", "Готово",
                                MessageBoxButtons.OK, MessageBoxIcon.Information);
            else
                MessageBox.Show("Не удалось добавить товар в корзину.", "Ошибка",
                                MessageBoxButtons.OK, MessageBoxIcon.Error);
        }

        // Обработка клика по карточке – открытие подробной информации
        private void OnProductClicked(object sender, Product product)
        {
            var detailPage = new ProductDetailsPage(product, currentUser);
            detailPage.BackRequested += (s, args) => ShowPage(catalogControl);
            ShowPage(detailPage);
        }

        private void ToggleMenu(object sender, EventArgs e)
        {
            menuControl.Visible = !menuControl.Visible;
        }

        private void Navigate(object sender, string pageKey)
        {
            switch (pageKey)
            {
                case "catalog":
                    ShowPage(catalogControl);
                    break;
                case "profile":
                    OpenProfileInternal();
                    break;
                case "about":
                    ShowPage(new AboutPage());
                    break;
                case "contacts":
                    ShowPage(new ContactsPage());
                    break;
            }
            menuControl.Visible = false;
        }

        private void OpenProfileInternal()
        {
            if (currentUser == null)
            {
                using (AuthForm auth = new AuthForm())
                {
                    if (auth.ShowDialog() == DialogResult.OK)
                        currentUser = auth.CurrentUser;
                    else
                        return;
                }
            }

            ProfilePage profilePage = new ProfilePage();
            profilePage.LogoutRequested += (s, args) =>
            {
                currentUser = null;
                ShowPage(catalogControl);
            };
            profilePage.LoadUser(currentUser);
            ShowPage(profilePage);
            menuControl.Visible = false;
        }

        private void OpenProfile(object sender, EventArgs e)
        {
            OpenProfileInternal();
        }

        private void OpenCart(object sender, EventArgs e)
        {
            if (currentUser == null)
            {
                MessageBox.Show("Чтобы открыть корзину, войдите в систему.", "Требуется авторизация",
                    MessageBoxButtons.OK, MessageBoxIcon.Information);
                return;
            }

            var cartPage = new CartPage(currentUser);
            // Подписка на событие завершения заказа
            cartPage.OrderCompleted += (s, args) =>
            {
                ShowPage(catalogControl); // возвращаемся в каталог
            };
            ShowPage(cartPage);
            menuControl.Visible = false;
        }

        private void ShowPage(UserControl page)
        {
            pnlContent.Controls.Clear();
            page.Dock = DockStyle.Fill;
            pnlContent.Controls.Add(page);
        }
    }
}
