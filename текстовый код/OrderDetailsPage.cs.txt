using System;
using System.Collections.Generic;
using System.Drawing;
using System.Linq;
using System.Windows.Forms;

namespace cond
{
    public partial class OrderDetailsPage : UserControl
    {
        private readonly int _orderId;
        private Order _order;
        private Panel _itemsContainer;
        private Label _lblTotal;
        private int _currentY;

        public event EventHandler BackRequested;

        public OrderDetailsPage(int orderId)
        {
            _orderId = orderId;
            InitializeComponent();
            LoadOrder();
            SetupUI();
            this.Load += (s, e) => PopulateItems();
        }

        private void LoadOrder()
        {
            _order = DatabaseHelper.GetOrderDetails(_orderId);
            if (_order == null)
            {
                MessageBox.Show("Заказ не найден.", "Ошибка", MessageBoxButtons.OK, MessageBoxIcon.Error);
                BackRequested?.Invoke(this, EventArgs.Empty);
            }
        }

        private void SetupUI()
        {
            this.Dock = DockStyle.Fill;
            this.BackColor = ThemeColors.Background;
            this.Padding = new Padding(0, 120, 0, 0); // отступ под главную шапку

            // Верхняя панель с заголовком
            var titlePanel = new Panel
            {
                Dock = DockStyle.Top,
                Height = 70,
                BackColor = Color.Transparent
            };
            var lblTitle = new Label
            {
                Text = $"Заказ №{_orderId}",
                Font = new Font("Segoe UI", 26, FontStyle.Bold),
                ForeColor = ThemeColors.Text,
                Location = new Point(30, 15),
                AutoSize = true
            };
            titlePanel.Controls.Add(lblTitle);
            this.Controls.Add(titlePanel);

            // Панель информации о заказе
            var infoPanel = new Panel
            {
                Dock = DockStyle.Top,
                AutoSize = true,
                BackColor = Color.Transparent,
                Padding = new Padding(30, 0, 30, 10)
            };

            // Дата
            var lblDate = new Label
            {
                Text = $"Дата: {_order.OrderDate:dd.MM.yyyy HH:mm}",
                Font = new Font("Segoe UI", 12),
                ForeColor = ThemeColors.Text,
                Dock = DockStyle.Top,
                Height = 25
            };
            infoPanel.Controls.Add(lblDate);

            // Статус
            var lblStatus = new Label
            {
                Text = $"Статус: {_order.Status}",
                Font = new Font("Segoe UI", 12, FontStyle.Bold),
                ForeColor = GetStatusColor(_order.Status),
                Dock = DockStyle.Top,
                Height = 25
            };
            infoPanel.Controls.Add(lblStatus);

            // Адрес доставки
            var lblAddress = new Label
            {
                Text = $"Адрес доставки: {_order.DeliveryAddress}",
                Font = new Font("Segoe UI", 12),
                ForeColor = ThemeColors.Text,
                Dock = DockStyle.Top,
                Height = 25,
                AutoEllipsis = true
            };
            infoPanel.Controls.Add(lblAddress);

            // Комментарий (если есть)
            if (!string.IsNullOrWhiteSpace(_order.Comment))
            {
                var lblComment = new Label
                {
                    Text = $"Комментарий: {_order.Comment}",
                    Font = new Font("Segoe UI", 12),
                    ForeColor = ThemeColors.Text,
                    Dock = DockStyle.Top,
                    Height = 25,
                    AutoEllipsis = true
                };
                infoPanel.Controls.Add(lblComment);
            }
            this.Controls.Add(infoPanel);

            // Небольшой отступ после информации о заказе
            this.Controls.Add(new Panel { Dock = DockStyle.Top, Height = 10, BackColor = Color.Transparent });

            // Нижняя панель с итогом и кнопкой "Назад"
            var bottomPanel = new Panel
            {
                Dock = DockStyle.Bottom,
                Height = 80,
                BackColor = Color.Transparent
            };

            var total = _order.TotalAmount;
            _lblTotal = new Label
            {
                Text = $"Итого: {total:F2} ₽",
                Font = new Font("Segoe UI", 18, FontStyle.Bold),
                ForeColor = ThemeColors.Accent,
                AutoSize = true,
                Location = new Point(30, 25)
            };

            var btnBack = new Button
            {
                Text = "← Назад",
                Font = new Font("Segoe UI", 14, FontStyle.Bold),
                FlatStyle = FlatStyle.Flat,
                BackColor = Color.LightGray,
                ForeColor = ThemeColors.Text,
                Cursor = Cursors.Hand,
                FlatAppearance = { BorderSize = 0 },
                Size = new Size(150, 50),
                Location = new Point(bottomPanel.Width - 180, 15),
                Anchor = AnchorStyles.Right | AnchorStyles.Top
            };
            btnBack.Click += (s, e) => BackRequested?.Invoke(this, EventArgs.Empty);

            bottomPanel.Controls.Add(_lblTotal);
            bottomPanel.Controls.Add(btnBack);
            this.Controls.Add(bottomPanel);

            // Центральная прокручиваемая область для списка товаров
            var scrollPanel = new Panel
            {
                Dock = DockStyle.Fill,
                AutoScroll = true,
                BackColor = Color.Transparent
            };

            _itemsContainer = new Panel
            {
                AutoSize = true,
                AutoSizeMode = AutoSizeMode.GrowAndShrink,
                BackColor = Color.Transparent
            };
            scrollPanel.Controls.Add(_itemsContainer);
            this.Controls.Add(scrollPanel);
        }

        private void PopulateItems()
        {
            if (_order?.Items == null) return;

            _itemsContainer.Controls.Clear();
            _currentY = 70; // увеличенный отступ сверху

            foreach (var item in _order.Items)
            {
                var card = CreateOrderItemCard(item);
                card.Location = new Point(10, _currentY);
                _itemsContainer.Controls.Add(card);
                _currentY += card.Height + 10;
            }
        }

        private Panel CreateOrderItemCard(OrderItem item)
        {
            int cardWidth = Math.Max(600, this.ClientSize.Width - 60);
            var card = new Panel
            {
                Width = cardWidth,
                Height = 90,
                BackColor = Color.White,
                Margin = new Padding(0)
            };

            card.Paint += (s, e) =>
            {
                using (var pen = new Pen(Color.LightGray, 1))
                    e.Graphics.DrawRectangle(pen, 0, 0, card.Width - 1, card.Height - 1);
            };

            // Название товара
            var lblName = new Label
            {
                Text = item.ProductName,
                Font = new Font("Segoe UI", 11, FontStyle.Bold),
                ForeColor = ThemeColors.Text,
                Location = new Point(10, 10),
                AutoSize = true
            };
            card.Controls.Add(lblName);

            // Количество × цена за единицу
            var lblDetails = new Label
            {
                Text = $"{item.Quantity} шт. × {item.PriceAtTime:F2} ₽",
                Font = new Font("Segoe UI", 9),
                ForeColor = Color.Gray,
                Location = new Point(10, 35),
                AutoSize = true
            };
            card.Controls.Add(lblDetails);

            // Общая стоимость позиции
            var lblSum = new Label
            {
                Text = $"{(item.Quantity * item.PriceAtTime):F2} ₽",
                Font = new Font("Segoe UI", 12, FontStyle.Bold),
                ForeColor = ThemeColors.Accent,
                AutoSize = true,
                Location = new Point(card.Width - 120, 20)
            };
            card.Controls.Add(lblSum);

            card.Resize += (s, e) =>
            {
                lblSum.Location = new Point(card.Width - 120, 20);
            };

            return card;
        }

        private Color GetStatusColor(string status)
        {
            switch (status?.ToLower())
            {
                case "новый": return Color.Blue;
                case "оплачен": return Color.Green;
                case "готов": return Color.Orange;
                case "выдан": return Color.Gray;
                default: return Color.Black;
            }
        }

        protected override void OnResize(EventArgs e)
        {
            base.OnResize(e);
            if (_itemsContainer != null && _itemsContainer.Controls.Count > 0)
                PopulateItems(); // перестраиваем карточки при изменении размера окна
        }
    }
}
