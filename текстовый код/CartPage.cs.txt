using System;
using System.Collections.Generic;
using System.Drawing;
using System.Linq;
using System.Windows.Forms;

namespace cond
{
    public partial class CartPage : UserControl
    {
        private User _currentUser;
        private FlowLayoutPanel flowPanel;
        private Label lblTotal;
        private Button btnCheckout;
        private List<CartItem> _cartItems;

        public CartPage(User currentUser)
        {
            _currentUser = currentUser ?? throw new ArgumentNullException(nameof(currentUser));
            InitializeComponent();
            SetupUI();
            LoadCart();
        }

        private void SetupUI()
        {
            this.Dock = DockStyle.Fill;
            this.BackColor = ThemeColors.Background;
            this.AutoScroll = true;
            this.Padding = new Padding(0, 120, 0, 0); // отступ под шапку

            // Основной контейнер
            var container = new TableLayoutPanel
            {
                Dock = DockStyle.Fill,
                ColumnCount = 1,
                RowCount = 3,
                Padding = new Padding(20),
                BackColor = Color.Transparent
            };
            container.RowStyles.Add(new RowStyle(SizeType.Absolute, 60)); // заголовок
            container.RowStyles.Add(new RowStyle(SizeType.Percent, 100)); // список товаров
            container.RowStyles.Add(new RowStyle(SizeType.Absolute, 80)); // итог и кнопка

            // Заголовок
            var lblTitle = new Label
            {
                Text = "Корзина",
                Font = new Font("HONOR Sans", 28, FontStyle.Bold),
                ForeColor = ThemeColors.Text,
                TextAlign = ContentAlignment.MiddleCenter,
                Dock = DockStyle.Fill
            };
            container.Controls.Add(lblTitle, 0, 0);

            // Панель для товаров с прокруткой
            var scrollPanel = new Panel
            {
                Dock = DockStyle.Fill,
                AutoScroll = true,
                BackColor = Color.Transparent
            };
            flowPanel = new FlowLayoutPanel
            {
                FlowDirection = FlowDirection.TopDown,
                WrapContents = false,
                AutoSize = true,
                AutoSizeMode = AutoSizeMode.GrowAndShrink,
                BackColor = Color.Transparent
            };
            scrollPanel.Controls.Add(flowPanel);
            container.Controls.Add(scrollPanel, 0, 1);

            // Нижняя панель с итогом и кнопкой
            var bottomPanel = new Panel { Dock = DockStyle.Fill, BackColor = Color.Transparent };
            var totalPanel = new FlowLayoutPanel
            {
                FlowDirection = FlowDirection.LeftToRight,
                WrapContents = false,
                AutoSize = true,
                BackColor = Color.Transparent
            };
            lblTotal = new Label
            {
                Text = "Итого: 0 ₽",
                Font = new Font("HONOR Sans", 18, FontStyle.Bold),
                ForeColor = ThemeColors.Accent,
                AutoSize = true
            };
            btnCheckout = new Button
            {
                Text = "Оформить заказ",
                Font = new Font("HONOR Sans", 16, FontStyle.Bold),
                FlatStyle = FlatStyle.Flat,
                BackColor = ThemeColors.Accent,
                ForeColor = ThemeColors.Text,
                Cursor = Cursors.Hand,
                FlatAppearance = { BorderSize = 0 },
                Size = new Size(220, 50),
                Margin = new Padding(20, 0, 0, 0)
            };
            btnCheckout.Click += BtnCheckout_Click;
            totalPanel.Controls.Add(lblTotal);
            totalPanel.Controls.Add(btnCheckout);
            bottomPanel.Controls.Add(totalPanel);
            totalPanel.Location = new Point(
                (bottomPanel.Width - totalPanel.Width) / 2,
                (bottomPanel.Height - totalPanel.Height) / 2
            );
            bottomPanel.Resize += (s, e) =>
            {
                totalPanel.Location = new Point(
                    (bottomPanel.Width - totalPanel.Width) / 2,
                    (bottomPanel.Height - totalPanel.Height) / 2
                );
            };
            container.Controls.Add(bottomPanel, 0, 2);

            this.Controls.Add(container);
        }

        private void LoadCart()
        {
            _cartItems = DatabaseHelper.GetCart(_currentUser.Id);
            flowPanel.Controls.Clear();

            if (_cartItems.Count == 0)
            {
                var lblEmpty = new Label
                {
                    Text = "Корзина пуста",
                    Font = new Font("HONOR Sans", 18),
                    ForeColor = Color.Gray,
                    AutoSize = true,
                    Margin = new Padding(10)
                };
                flowPanel.Controls.Add(lblEmpty);
                btnCheckout.Enabled = false;
            }
            else
            {
                foreach (var item in _cartItems)
                {
                    var itemControl = CreateCartItemControl(item);
                    flowPanel.Controls.Add(itemControl);
                }
                btnCheckout.Enabled = true;
            }
            UpdateTotal();
        }

        private Control CreateCartItemControl(CartItem item)
        {
            var panel = new Panel
            {
                Width = flowPanel.Width - 40,
                Height = 100,
                BackColor = Color.White,
                Margin = new Padding(10)
            };
            panel.Paint += (s, e) =>
            {
                ControlPaint.DrawBorder(e.Graphics, panel.ClientRectangle,
                    Color.LightGray, 1, ButtonBorderStyle.Solid,
                    Color.LightGray, 1, ButtonBorderStyle.Solid,
                    Color.LightGray, 1, ButtonBorderStyle.Solid,
                    Color.LightGray, 1, ButtonBorderStyle.Solid);
            };

            // Картинка
            var pb = new PictureBox
            {
                Size = new Size(80, 80),
                Location = new Point(10, 10),
                SizeMode = PictureBoxSizeMode.Zoom,
                Image = LoadImage(item.Product.ImageFilename)
            };
            panel.Controls.Add(pb);

            // Название
            var lblName = new Label
            {
                Text = item.Product.Name,
                Font = new Font("HONOR Sans", 12, FontStyle.Bold),
                Location = new Point(100, 15),
                AutoSize = true
            };
            panel.Controls.Add(lblName);

            // Цена за единицу
            var lblPrice = new Label
            {
                Text = $"{item.Product.Price:F2} ₽",
                Font = new Font("HONOR Sans", 10),
                ForeColor = Color.Gray,
                Location = new Point(100, 45),
                AutoSize = true
            };
            panel.Controls.Add(lblPrice);

            // Количество
            var lblQty = new Label
            {
                Text = $"Кол-во: {item.Quantity}",
                Font = new Font("HONOR Sans", 10),
                Location = new Point(100, 65),
                AutoSize = true
            };
            panel.Controls.Add(lblQty);

            // Сумма
            var lblSum = new Label
            {
                Text = $"{(item.Product.Price * item.Quantity):F2} ₽",
                Font = new Font("HONOR Sans", 12, FontStyle.Bold),
                ForeColor = ThemeColors.Accent,
                Location = new Point(panel.Width - 120, 40),
                AutoSize = true
            };
            panel.Controls.Add(lblSum);

            // Кнопки управления
            var btnPlus = new Button
            {
                Text = "+",
                Size = new Size(30, 30),
                Location = new Point(panel.Width - 80, 10),
                FlatStyle = FlatStyle.Flat,
                BackColor = ThemeColors.Accent,
                ForeColor = Color.White,
                Cursor = Cursors.Hand
            };
            btnPlus.Click += (s, e) => ChangeQuantity(item, 1);
            panel.Controls.Add(btnPlus);

            var btnMinus = new Button
            {
                Text = "-",
                Size = new Size(30, 30),
                Location = new Point(panel.Width - 80, 45),
                FlatStyle = FlatStyle.Flat,
                BackColor = Color.LightGray,
                ForeColor = Color.Black,
                Cursor = Cursors.Hand,
                Enabled = item.Quantity > 1
            };
            btnMinus.Click += (s, e) => ChangeQuantity(item, -1);
            panel.Controls.Add(btnMinus);

            var btnRemove = new Button
            {
                Text = "Удалить",
                Size = new Size(70, 25),
                Location = new Point(panel.Width - 80, 80),
                FlatStyle = FlatStyle.Flat,
                BackColor = Color.FromArgb(255, 100, 100),
                ForeColor = Color.White,
                Cursor = Cursors.Hand,
                Font = new Font("HONOR Sans", 8)
            };
            btnRemove.Click += (s, e) => RemoveItem(item);
            panel.Controls.Add(btnRemove);

            panel.Resize += (s, e) =>
            {
                // адаптация позиций при изменении ширины
                lblSum.Location = new Point(panel.Width - 120, 40);
                btnPlus.Location = new Point(panel.Width - 80, 10);
                btnMinus.Location = new Point(panel.Width - 80, 45);
                btnRemove.Location = new Point(panel.Width - 80, 80);
            };

            return panel;
        }

        private void ChangeQuantity(CartItem item, int delta)
        {
            int newQty = item.Quantity + delta;
            if (newQty < 1) return;

            if (DatabaseHelper.UpdateCartItemQuantity(_currentUser.Id, item.ProductId, newQty))
            {
                item.Quantity = newQty;
                LoadCart(); // перезагружаем для обновления отображения
            }
        }

        private void RemoveItem(CartItem item)
        {
            if (DatabaseHelper.RemoveFromCart(_currentUser.Id, item.ProductId))
            {
                LoadCart();
            }
        }

        private void UpdateTotal()
        {
            decimal total = _cartItems.Sum(item => item.Product.Price * item.Quantity);
            lblTotal.Text = $"Итого: {total:F2} ₽";
        }

        private Image LoadImage(string filename)
        {
            if (string.IsNullOrEmpty(filename))
                return CreateDefaultImage();

            string path = System.IO.Path.Combine(Application.StartupPath, "Images", filename);
            if (System.IO.File.Exists(path))
            {
                try
                {
                    using (var fs = new System.IO.FileStream(path, System.IO.FileMode.Open, System.IO.FileAccess.Read))
                        return Image.FromStream(fs);
                }
                catch { }
            }
            return CreateDefaultImage();
        }

        private Image CreateDefaultImage()
        {
            Bitmap bmp = new Bitmap(80, 80);
            using (Graphics g = Graphics.FromImage(bmp))
            {
                g.Clear(Color.LightPink);
                g.DrawString("🍰", new Font("HONOR Sans", 30), Brushes.White, 20, 20);
            }
            return bmp;
        }

        private void BtnCheckout_Click(object sender, EventArgs e)
        {
            // Переход к странице оформления заказа
            var parent = this.Parent as Panel; // pnlContent
            if (parent != null)
            {
                var checkoutPage = new CheckoutPage(_currentUser, _cartItems);
                checkoutPage.OrderCompleted += (s, args) =>
                {
                    // После завершения заказа возвращаемся в каталог
                    parent.Controls.Clear();
                    parent.Controls.Add(new CatalogControl()); // предполагаем, что CatalogControl создаётся заново (или используем существующий)
                };
                parent.Controls.Clear();
                parent.Controls.Add(checkoutPage);
            }
        }
    }
}
